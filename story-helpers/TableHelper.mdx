<!--
This is a utility component for creating tables in Storybook MDX
documentation. By defining it as an MDX component, Storybook docs
will apply its built in styles to the table. More info about using
MDX can be found on their website: https://mdxjs.com/docs/using-mdx/.

@example
import Table from ./TableHelper.mdx;

<Table
    headings={['Heading 1', 'Heading 2', 'Heading 3']}
    body={[
        ['Data for row 1, column 1', 'Data for row 1, column 2', 'Data for row 1, column 3'],
        ['Data for row 2, column 1', 'Data for row 2, column 2', 'Data for row 2, column 3'],
    ]}
    columnAignment={[null, 'right', 'center']}
/>
-->

<table>
    <thead>
        <tr>
            {props.headings.map((heading) => (
                // eslint-disable-next-line jsx-a11y/control-has-associated-label
                <th
                    key={heading}
                    dangerouslySetInnerHTML={{ __html: heading }}
                />
            ))}
        </tr>
    </thead>
    <tbody>
        {props.body.map((row) => (
            <tr key={row[0]}>
                {row.map((cell, index) => (
                    <td
                        key={cell}
                        align={props.columnAlignment?.[index] || null}
                        dangerouslySetInnerHTML={{ __html: cell }}
                    />
                ))}
            </tr>
        ))}
    </tbody>
</table>
