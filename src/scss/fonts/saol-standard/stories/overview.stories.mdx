# Saol Standard

Standard font version of the Saol family.

## Import Font

Import the Saol Standard font styles into one of your SASS modules.

```css
// _fonts.scss

@import 'mfe-core/typography/saol-standard.css';
```

## Use font family

```css
.element {
    font-family: 'Saol Standard';
}
```

export function Template(args) {
    const { weights, size, text, fontFamily, italic } = args;
    return (
        <FontHelper
            text={text}
            size={size}
            fontFamily={fontFamily}
            weights={weights}
            italic={italic}
        />
    );
}
