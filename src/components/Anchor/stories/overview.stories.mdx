## Use example

```jsx
import Anchor from 'mfe-core/ui/Anchor';

const href = 'https://twitter.com';
const openInNewTab = true;
const clickHandler = () => console.log('hello world');
const label = 'Go to twitter';
const linkRef = useRef(null);

return (
    <Anchor
        className={styles.link}
        href={href}
        label={label}
        newTab={openInNewTab}
        onClick={clickHandler}
        ref={linkRef}
        tabIndex="0"
    >
        <img src="twitter.svg" />
    </Anchor>
);
```
