# Default Anchor

By default the `Anchor` opens the URL in same tab.
And the default `label` tag is `null`.

<Meta
    title="Components/Anchor/Features"
    component={Anchor}
    argTypes={{
        children: { table: { disable: true } },

        label: { table: { disable: true } },
        onClick: { table: { disable: true } },
        tabIndex: { table: { disable: true } },
        newTab: { table: { disable: true } },
    }}
/>

export function Template(args) {
    /* eslint-disable no-unused-vars */
    const { href, children, label, newTab, onClick, tabIndex, ref } = args;
    return (
        <Anchor
            href={href}
            label={label}
            newTab={newTab}
            onClick={onClick}
            tabIndex={tabIndex}
        >
            {children}
        </Anchor>
    );
}
