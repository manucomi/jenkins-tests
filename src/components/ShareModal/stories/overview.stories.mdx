## Use example

```jsx
import ShareModal from 'mfe-core/ui/ShareModal';

const article = {
    title: 'A Demo on Sharing',
    type: 'Digital Article',
    url: 'https://hbr.org/article/demo-on-sharing',
};

const [isShareOpen, setIsShareOpen] = useState(false);

return (
    <>
        <button onClick={() => setIsShareOpen(true)} type="button">
            Share
        </button>
        {isShareOpen && (
            <ShareModal
                onClose={() => setIsShareOpen(false)}
                shareData={article}
                shareOnFacebook={FBcallback}
                shareOnTwitter={TWcallback}
                shareOnLinkedin={LinkedInCallback}
            />
        )}
    </>
);
```
