import { Meta, Story, Canvas } from '@storybook/addon-docs';
import { useState } from 'react';
import classNames from 'classnames';
import SaveModal from 'Components/SaveModal/SaveModal';
import styles from './stories.module.scss';

# Playground

Play with the controls and see how the component changes.

<Meta title="Components/SaveModal" component={SaveModal} />

export function Template(args) {
    const { dataPoints, userEmailAddress } = args;
    const [isSaveOpen, setIsSaveOpen] = useState(false);
    const handleClose = () => setIsSaveOpen(false);
    const handleOpen = () => setIsSaveOpen(true);
    const buttonClasses = classNames({
        [styles.btn]: true,
        [styles.text]: true,
    });
    return (
        <>
            <button
                onClick={handleOpen}
                className={buttonClasses}
                type="button"
            >
                Click to open Save Modal
            </button>
            {isSaveOpen && (
                <SaveModal
                    onClose={handleClose}
                    dataPoints={dataPoints}
                    userEmailAddress={userEmailAddress}
                />
            )}
        </>
    );
}

<Canvas>
    <Story
        name="Playground"
        args={{
            dataPoints: { id: 123, title: 'Learn the Alphabet' },
            userEmailAddress: 'person@domain.com',
        }}
    >
        {Template.bind({})}
    </Story>
</Canvas>
