## Usage example

```jsx
import SaveModal from 'mfe-core/ui/SaveModal';

const [isSaveOpen, setIsSaveOpen] = useState(false);
const handleClose = () => setIsSaveOpen(false);
const handleOpen = () => setIsSaveOpen(true);
const logData = { id: 123, title: 'Learn the Alphabet' };
return (
    <>
       <button onClick={handleOpen}>Open</button>
       {isSaveOpen && (
           <SaveModal
               onClose={handleClose}
               dataPoints={logData}
               userEmailAddress={userEmailAddress}
           />
        )}
     </>;
);
```
